CREATE TABLE RESERVE
(
ResTime		DATETIME NOT NULL DEFAULT SYSDATE PRIMARY KEY,
STATUS		CHAR (1) NOT NULL DEFAULT "NO",
ResNum		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
LibID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
DocID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
CopyNum		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,	
);

create table USER 
(
SSN INT (9) PRIMARY KEY, User_Type_Admin char(1) NOT NULL, 
LoginName Varchar(100) NOT NULL, Password Varchar(20) NOT NULL, 
First_Name Varchar(100) NOT NULL, Last_Name Varchar(100) NOT NULL
);

CREATE TABLE Library_Card
(
Address			VARCHAR (100) NOT NULL,
PhoneNum		NUMBER UNSIGNED NOT NULL,
RegistrationID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
L_Name			VARCHAR (100) NOT NULL,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			DocID FOREIGN KEY REFERENCES DOC,
LibID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			LibID FOREIGN KEY REFERENCES BRANCH,
UserType		VARCHAR (100) NOT NULL,
PublisherID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			PublisherID FOREIGN KEY REFERENCES PUBLISHER,
);

CREATE TABLE DOC
(
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
Publisher		VARCHAR (100) NOT NULL,
Title			VARCHAR (100) NOT NULL,
PublisherID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			PublisherID FOREIGN KEY REFERENCES PUBLISHER,
LibID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			LibID FOREIGN KEY REFERENCES BRANCH,
);

CREATE TABLE RENT
(
PublisherID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			PublisherID FOREIGN KEY REFERENCES PUBLISHER,
RentDateTime		DATETIME NOT NULL DEFAULT SYSDATE,
LibCardID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
ResNum			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
LibID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
CopyNum			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
);

CREATE TABLE PUBLISHER
(
PublisherID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
PubAdd			VARCHAR (100) NOT NULL,
PubDate			DATETIME NOT NULL,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			DocID FOREIGN KEY REFERENCES DOC,
LibID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			LibID FOREIGN KEY REFERENCES BRANCH,

);

CREATE TABLE BRANCH
(
LibID			NUMBER NOT NULL	PRIMARY KEY,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			DocID FOREIGN KEY REFERENCES DOC,
PublisherID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			PublisherID FOREIGN KEY REFERENCES PUBLISHER,
LibAdd			VARCHAR (100) NOT NULL,
LibName			VARCHAR (100) NOT NULL,
);


CREATE TABLE CHIEF_EDITOR
(
ChiefEdID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
AuthorID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			AuthorID FOREIGN KEY REFERENCES BOOK,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
IssueNum		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
ChiefName		VARCHAR (100) NOT NULL,
);

CREATE TABLE BOOK
(
AuthorID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
ChiefEdID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			ChiefEdID FOREIGN KEY REFERENCES CHIEF_EDITOR,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
Auth_Name		VARCHAR (100) NOT NULL,
ISBN			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
);


CREATE TABLE JOURNAL_VOL
(
IssueNum		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
JournalNum		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
ChiefEdID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			ChiefEdID FOREIGN KEY REFERENCES CHIEF_EDITOR,
AuthorID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			AuthorID FOREIGN KEY REFERENCES BOOK,
);

CREATE TABLE JOURNAL_ISSUE
(
IssueNum		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT CONSTRAINT 
			IssueNum FOREIGN KEY REFERENCES JOURNAL_vOL,
DocID			NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
ChiefEdID		NUMBER UNSIGNED NOT NULL AUTO_INCREMENT,
L_Name			VARCHAR (100) NOT NULL,
);


COMMIT;
